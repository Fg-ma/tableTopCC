
# Config
nano ~/.vnc/xstartup
chmod +x ~/.vnc/xstartup

#!/bin/sh
unset SESSION_MANAGER
unset DBUS_SESSION_BUS_ADDRESS

# Load Xresources if present
[ -f "$HOME/.Xresources" ] && xrdb "$HOME/.Xresources"

# Start GNOME on Xorg
exec gnome-session


# Start up command
vncserver :8 -localhost yes -xstartup ~/.vnc/xstartup

# Reset nftables permission
sudo nft -a list chain inet filter input
sudo nft delete rule inet filter input handle 36
sudo nft add rule inet filter input ip saddr 10.136.. tcp dport 22 ct state new accept
sudo sh -c "nft list ruleset > /etc/nftables.conf"
sudo systemctl restart nftables
